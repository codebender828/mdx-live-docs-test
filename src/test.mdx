#  `Demo` example

<LiveEditor>
{`
<template>
  <div>
    <h3>Count: {{ count }} </h3>
    <button @click="increment">
      Increase
    </button>
    <button @click="decrement">
      Decrease
    </button>
  </div>
</template>
<script>
  export default {
    name: 'Count',
    data () {
      return {
        count: 12
      }
    },
    methods: {
      increment () {
        this.count++
      },
      decrement () {
        this.count--
      }
    }
  }
<\/script>
`}
</LiveEditor>
